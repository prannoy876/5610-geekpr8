About the Experiment


       In this experiments we create a Food service that contains
       nutrition content certain foods.

       Using $routeParams we can send information from one view to another.This parameter
       can be accessed by that particular views controller.

       Pay special attention to the FoodInfoCtrl controller and see how is recieves
       the $routeParams which it then uses to generate nutrition info for the 
       requested food item

Javascript


       app.config(['$routeProvider',
    function ($routeProvider) {
        $routeProvider.
            when('/', {
                templateUrl: 'partials/foods.html',
                controller: 'FoodCtrl'
            }).
            when('/foodInfo/:food', {
                templateUrl: 'partials/foodinfo.html',
                controller: 'FoodInfoCtrl'
            }).
            otherwise({
                redirectTo : '/'
            });
    }]);

app.controller("FoodCtrl", function ($scope, FoodInformation) {
    $scope.foodInfo = FoodInformation.getFoods();
});

app.controller("FoodInfoCtrl", function ($scope, $routeParams, FoodInformation) {
    $scope.food = $routeParams.food;
    var foodDetails = FoodInformation.getFoodInfo($routeParams.food);
    $scope.food = foodDetails.nutritionFacts;
});
