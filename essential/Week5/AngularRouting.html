About the Experiment


           In this Experiment we demonstrate the
           use of routing in Angular JS to move
           from one view to another.
           
           Routing makes loading a Web Page faster
           and makes it very easy to modularise
           the web application.
           
           In this experiment we just move from 
           one page to the next.

 var app = angular.module("RoutApp", ['ngRoute']);

app.factory('Data', function(){

return [{ name: "Picard", description: "Captain" },
{ name: "Riker", description: "NUmber One" },
{ name: "Worf", description: "Security"}];

  
});



app.config(['$routeProvider',
  function ($routeProvider) {
      $routeProvider.
      when('/', {
          templateUrl: 'partials/list.html',
          controller: 'myController'
      }).
      when('/new', {
          templateUrl: 'partials/add.html',
          controller: 'myController2'
      }).
      otherwise({
          redirectTo: '/'
      });
  } ]);
