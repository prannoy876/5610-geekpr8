About the Experiment

     In this experiment we create a RESTFUL API service
     using Node.js to perform the basic CRUD operations
     which make changes to Data living on the server.

     The above experiment is hosted remotely using openshift
     and is not available on the Northeastern server.

     We make use of get,put,post and delete HTTP requests
     to manipulate the data living on the server.

Javascript

var app = angular.module("CrudApp", []);


app.controller("CrudCtrl", function ($scope, $http) {
    $http.get("/flights")
    .success(function (response) {

        $scope.flights = response;
    });

    $scope.remove=function(index){
      $http.delete("/flights/"+ index)
    .success(function (response) {

        $scope.flights = response;
    });
    };

    $scope.selectedIndex=null;
    
     $scope.edit=function(index){
        $scope.selectedIndex=index;
        $scope.flight=$scope.flights[index];
    };

     $scope.save=function(flight){
        
         $http.put("/flight/" + $scope.selectedIndex,flight)
    .success(function (response) {

        $scope.flights = response;
    });
    };

     $scope.add=function(flight){
      $http.post("/flight",flight)
    .success(function (response) {

        $scope.flights = response;
    });
    };

});
